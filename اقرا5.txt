```python
# -*- coding: utf-8 -*-
"""
تعريف استثناءات اللغة 'اقرأ' وأدوات إدارة سياق التنفيذ (context).
يوفّر تتبعًا مبسطًا لسياق العقد أثناء التنفيذ لرسائل أخطاء أكثر فائدة.
"""

from dataclasses import dataclass
import threading
import traceback

_local = threading.local()

def _get_ctx_stack():
    if not hasattr(_local, "ctx_stack"):
        _local.ctx_stack = []
    return _local.ctx_stack

def push_context(ctx: str):
    _get_ctx_stack().append(ctx)

def pop_context():
    stack = _get_ctx_stack()
    if stack:
        stack.pop()

def current_context():
    return list(_get_ctx_stack())

@dataclass
class TraceEntry:
    message: str

class خطأاقرأ(Exception):
    def __init__(self, رسالة, سطر=None, عمود=None, *, cause=None, include_traceback=False):
        """
        رسالة: نص الرسالة (عربي)
        سطر/عمود: تفاصيل موضعية إن وُجدت
        cause: استثناء أصلي (اختياري)
        include_traceback: لو True نحفظ تتبُّع بايثون الأصلي كنص للمساعدة في التصحيح
        """
        self.رسالة = رسالة
        self.سطر = سطر
        self.عمود = عمود
        self._context = current_context()
        self._cause = cause
        self._py_tb = None
        if include_traceback and cause is not None:
            try:
                self._py_tb = traceback.format_exc()
            except Exception:
                self._py_tb = None

        if سطر is not None and عمود is not None:
            super().__init__(f"{رسالة} (في السطر {سطر}، العمود {عمود})")
        elif سطر is not None:
            super().__init__(f"{رسالة} (في السطر {سطر})")
        else:
            super().__init__(رسالة)

    def with_source(self, مصدر: str):
        if self.سطر is None:
            return str(self)
        lines = مصدر.splitlines()
        if 1 <= self.سطر <= len(lines):
            line_text = lines[self.سطر - 1]
            col = self.عمود or 1
            if col < 1: col = 1
            if col > len(line_text) + 1: col = len(line_text) + 1
            pointer = ' ' * (col - 1) + '^'
            ctx = self.format_context()
            return f"{self.رسالة} (في السطر {self.سطر}، العمود {col})\n{line_text}\n{pointer}\n{ctx}"
        return str(self)

    def format_context(self):
        if not self._context:
            return ""
        lines = ["--- سياق تنفيذ (مقلوب، أحدث أول):"]
        for i, c in enumerate(reversed(self._context), 1):
            lines.append(f"  {i}. {c}")
        if self._py_tb:
            lines.append("--- تتبُّع بايثون الأصلي ---")
            lines.append(self._py_tb)
        return "\n".join(lines)

    def __str__(self):
        base = self.رسالة
        if self.سطر is not None:
            base = f"{base} (سطر {self.سطر})"
        ctx = self.format_context()
        if ctx:
            return f"{base}\n{ctx}"
        return base
```

# ملف: `iqra/interpreter.py`

```python
# -*- coding: utf-8 -*-
from . import ast_nodes as ast
from .errors import خطأاقرأ, push_context, pop_context
from . import tokens
import random

class _ReturnSignal(Exception):
    def __init__(self, value):
        self.value = value

class Interpreter:
    """
    مفسّر مبسّط يدعم:
    - تعريف واستدعاء الدوال
    - استدعاء غير مباشر (assign function to var ثم call)
    - closures (التقاط مرجعي لبيئة التعريف)
    - رسائل خطأ أوضح مع سياق العقد AST
    """
    def __init__(self, builtins=None):
        self.frames = [{}]  # stack frames; frames[0] = global
        self.functions = {}
        self.builtins = {
            'ادخل': lambda prompt='': input(prompt),
            'عشوائي': lambda a=0, b=1: random.randint(int(a), int(b)),
        }
        if builtins:
            self.builtins.update(builtins)

    def current_env(self):
        return self.frames[-1]

    def push_env(self):
        self.frames.append({})

    def pop_env(self):
        if len(self.frames) == 1:
            self.frames[-1].clear()
        else:
            self.frames.pop()

    def resolve_var(self, name):
        for env in reversed(self.frames):
            if name in env:
                return env[name]
        if name in self.functions:
            return self.functions[name]
        if name in self.builtins:
            return self.builtins[name]
        return None

    def store_var(self, name, val):
        for env in reversed(self.frames):
            if name in env:
                env[name] = val
                return
        self.current_env()[name] = val

    def interpret(self, node):
        push_context(f"Node:{node.__class__.__name__}")
        try:
            method = getattr(self, f'visit_{node.__class__.__name__}', None)
            if not method:
                raise خطأاقرأ(f'لا يوجد معالج للعقدة: {node.__class__.__name__}')
            return method(node)
        except خطأاقرأ:
            raise
        except Exception as e:
            raise خطأاقرأ(f"خطأ داخلي أثناء معالجة {node.__class__.__name__}: {e}", None, None, cause=e, include_traceback=True)
        finally:
            pop_context()

    # أمثلة من visiters الأساسية (يجب أن تتناغم مع ast_nodes)
    def visit_Program(self, node):
        for s in node.statements:
            self.interpret(s)

    def visit_PrintStatement(self, node):
        val = self.interpret(node.expression)
        print(val)

    def visit_StringLiteral(self, node):
        return node.value

    def visit_NumberLiteral(self, node):
        return node.value

    def visit_BooleanLiteral(self, node):
        return node.value

    def visit_ListLiteral(self, node):
        return [self.interpret(e) for e in node.elements]

    def visit_DictLiteral(self, node):
        d = {}
        for k_node, v_node in node.pairs:
            k = self.interpret(k_node)
            v = self.interpret(v_node)
            d[k] = v
        return d

    def _truthy(self, v):
        if isinstance(v, bool): return v
        if isinstance(v, (int, float)): return v != 0
        if isinstance(v, str): return v != ''
        if v is None: return False
        if isinstance(v, (list, dict)): return len(v) != 0
        return bool(v)

    def visit_BinaryOp(self, node):
        left = self.interpret(node.left)
        right = self.interpret(node.right)
        op = node.op
        if op == tokens.EQ: return left == right
        if op == tokens.NE: return left != right
        if op == tokens.LT: return left < right
        if op == tokens.GT: return left > right
        if op == tokens.LE: return left <= right
        if op == tokens.GE: return left >= right
        if op == tokens.AND: return self._truthy(left) and self._truthy(right)
        if op == tokens.OR:  return self._truthy(left) or self._truthy(right)
        if op == tokens.PLUS:
            if isinstance(left, list) and isinstance(right, list):
                return left + right
            if isinstance(left, str) or isinstance(right, str):
                return str(left) + str(right)
            return left + right
        if op == tokens.MINUS: return left - right
        if op == tokens.MULT: return left * right
        if op == tokens.DIV:
            if right == 0:
                raise خطأاقرأ('قسمة على صفر')
            return left / right
        raise خطأاقرأ(f'عامل غير مدعوم: {op}')

    def visit_UnaryOp(self, node):
        val = self.interpret(node.operand)
        op = node.op
        if op == tokens.NOT:
            return not self._truthy(val)
        if op == tokens.MINUS:
            return -val
        raise خطأاقرأ(f'عامل أحادي غير مدعوم: {op}')

    def visit_VarAssign(self, node):
        val = self.interpret(node.expression)
        self.store_var(node.name, val)
        return val

    def visit_SetItem(self, node):
        coll = self.interpret(node.collection)
        idx = self.interpret(node.index)
        val = self.interpret(node.expression)
        if isinstance(coll, list):
            if not isinstance(idx, int):
                raise خطأاقرأ('الفهرس يجب أن يكون عدداً صحيحاً')
            if idx < 0: idx = len(coll) + idx
            if idx < 0 or idx >= len(coll):
                raise خطأاقرأ('فهرس خارج النطاق')
            coll[idx] = val
            return val
        if isinstance(coll, dict):
            coll[idx] = val
            return val
        raise خطأاقرأ('النوع غير قابل للتعيين عبر فهرس')

    def visit_VarReference(self, node):
        val = self.resolve_var(node.name)
        if val is None:
            raise خطأاقرأ(f'متغير أو دالة غير معرفة: {node.name}')
        return val

    def visit_Index(self, node):
        collection = self.interpret(node.collection)
        idx = self.interpret(node.index)
        if isinstance(collection, (list, tuple, str)):
            n = len(collection)
            if not isinstance(idx, int):
                raise خطأاقرأ('الفهرس يجب أن يكون عدداً صحيحاً')
            if idx < 0: idx = n + idx
            if idx < 0 or idx >= n:
                raise خطأاقرأ('فهرس خارج النطاق')
            return collection[idx]
        if isinstance(collection, dict):
            return collection.get(idx, None)
        raise خطأاقرأ('نوع غير قابل للفهرسة')

    def visit_Slice(self, node):
        collection = self.interpret(node.collection)
        start = self.interpret(node.start) if node.start is not None else None
        stop  = self.interpret(node.stop)  if node.stop is not None else None
        if isinstance(collection, (list, str)):
            return collection[slice(start, stop)]
        raise خطأاقرأ('القطع (slice) مدعوم فقط للقوائم والسلاسل')

    def visit_FunctionDef(self, node):
        func_obj = {
            'type': 'userfunc',
            'ast': node,
            'params': list(node.params),
            'closure': self.current_env(),
        }
        self.functions[node.name] = func_obj
        return None

    def visit_FunctionCall(self, node):
        # determine callable object
        callable_obj = None
        if isinstance(node.name, str):
            name = node.name
            # var with this name?
            val = None
            for env in reversed(self.frames):
                if name in env:
                    val = env[name]; break
            if val is not None:
                callable_obj = val
            elif name in self.builtins:
                callable_obj = self.builtins[name]
            elif name in self.functions:
                callable_obj = self.functions[name]
            else:
                raise خطأاقرأ(f'دالة أو متغير باسم {name} غير معرفة')
        else:
            callable_obj = self.interpret(node.name)

        args_vals = [self.interpret(a) for a in node.args]

        # python callable
        if callable(callable_obj) and not (isinstance(callable_obj, dict) and callable_obj.get('type') == 'userfunc'):
            return callable_obj(*args_vals)

        # user function
        if isinstance(callable_obj, dict) and callable_obj.get('type') == 'userfunc':
            fn = callable_obj
            if len(args_vals) != len(fn.get('params', [])):
                raise خطأاقرأ(f'عدد الوسائط في استدعاء {fn["ast"].name} غير صالح')
            # push closure frame reference if provided
            if fn.get('closure') is not None:
                self.frames.append(fn['closure'])
            self.push_env()
            for n, v in zip(fn.get('params', []), args_vals):
                self.current_env()[n] = v
            try:
                for s in fn['ast'].body:
                    self.interpret(s)
            except _ReturnSignal as r:
                self.pop_env()
                if fn.get('closure') is not None:
                    self.frames.pop()
                return r.value
            self.pop_env()
            if fn.get('closure') is not None:
                self.frames.pop()
            return None

        raise خطأاقرأ('نوع قابل للاستدعاء غير معروف')

    def visit_ReturnStatement(self, node):
        val = self.interpret(node.expression) if node.expression is not None else None
        raise _ReturnSignal(val)

    def visit_IfStatement(self, node):
        cond = self.interpret(node.condition)
        if self._truthy(cond):
            for s in node.true_statements:
                self.interpret(s)
        else:
            for s in node.false_statements:
                self.interpret(s)

    def visit_WhileStatement(self, node):
        cond_val = self.interpret(node.condition)
        if isinstance(cond_val, (int, float)):
            for _ in range(int(cond_val)):
                for s in node.body:
                    self.interpret(s)
            return
        while self._truthy(self.interpret(node.condition)):
            for s in node.body:
                self.interpret(s)
```

```python
# -*- coding: utf-8 -*-
import pytest
from iqra.lexer import Lexer
from iqra.parser import Parser
from iqra.interpreter import Interpreter
from iqra.errors import خطأاقرأ

def run_src(src):
    lexer = Lexer(src)
    tokens = lexer.tokenize()
    parser = Parser(tokens)
    ast = parser.parse()
    interp = Interpreter()
    return interp.interpret(ast)

def test_assignment_and_arithmetic():
    src = """
بسم الله
x = 2
y = 3
اطبع x + y
"""
    run_src(src)

def test_closure_and_nonlocal_behaviour():
    src = """
بسم الله
دالة outer()
    x = 10
    دالة inner()
        x = x + 1
        ارجع x
    انتهى
    ارجع inner
انتهى

f = outer()
اطبع f()
"""
    run_src(src)

def test_indirect_call():
    src = """
بسم الله
دالة add(a,b)
    ارجع a + b
انتهى

fn = add
اطبع fn(4,5)
"""
    run_src(src)

def test_unknown_var_raises():
    src = """
بسم الله
اطبع unknown_var
"""
    with pytest.raises(خطأاقرأ):
        run_src(src)
```

```python
# -*- coding: utf-8 -*-
import pytest
from iqra.lexer import Lexer
from iqra.parser import Parser

from iqra.vm import VM
from iqra.stdlib import text as textlib, files as fileslib, date as datelib

def compile_and_run_vm(src):
    lexer = Lexer(src)
    tokens = lexer.tokenize()
    parser = Parser(tokens)
    ast = parser.parse()
    comp = Compiler()
    comp.compile(ast)
    code, consts = comp.code, comp.consts
    vm = VM(code, consts, globals_vars={}, builtins={
        'ادخل': lambda prompt='': input(prompt),
        'عشوائي': lambda a=0,b=1: __import__('random').randint(int(a), int(b)),
        'split': textlib.split,
        'read': fileslib.read,
        'today': datelib.today,
    }, enable_jit=False)
    vm.run()

def test_vm_call_simple():
    src = """
بسم الله
دالة add(a,b)
    ارجع a + b
انتهى
اطبع add(1,2)
"""
    compile_and_run_vm(src)

    src = """
بسم الله
دالة mul(a,b)
    ارجع a * b
انتهى
f = mul
اطبع f(3,4)
"""
    compile_and_run_vm(src)
```

# ملف: `iqra/errors.py`

أدرجت لك الملف كاملاً أدناه — هذا الإصدار مُدقّق، يحسّن رسائل الخطأ ويحتفظ بسياق تنفيذ (context stack) يساعد كثيرًا في تتبُّع مصدر الخطأ داخل AST أو أثناء تنفيذ المفسّر/الـVM.

```python
# -*- coding: utf-8 -*-
"""
تعريف استثناءات لغة "اقرأ" وأدوات إدارة سياق التنفيذ (context).
يوفّر تتبُّعًا مبسّطًا لسياق العقد أثناء التنفيذ لرسائل أخطاء أكثر فائدة.
"""

from dataclasses import dataclass
import threading
import traceback

_local = threading.local()

def _get_ctx_stack():
    """إرجاع الستاك المحلي الخاص بالخيط (thread-local)."""
    if not hasattr(_local, "ctx_stack"):
        _local.ctx_stack = []
    return _local.ctx_stack

def push_context(ctx: str):
    """دفع عنصر سياق جديد إلى الستاك (عادةً اسم العقدة أو وصف مختصر)."""
    _get_ctx_stack().append(ctx)

def pop_context():
    """نزْع أحدث عنصر سياق إذا وُجد."""
    stack = _get_ctx_stack()
    if stack:
        stack.pop()

def current_context():
    """إرجاع نسخة من الستاك الحالي (قائمة)."""
    return list(_get_ctx_stack())

@dataclass
class TraceEntry:
    message: str

class خطأاقرأ(Exception):
    """
    استثناء عام للغة اقرأ.
    الحقول:
      - رسالة (عربي)
      - سطر، عمود (اختياريان) لتحديد الموقع في مصدر النص إن وُجد
      - _context: لقطة من ستاك السياق أثناء إنشاء الخطأ
      - _py_tb: تتبُّع بايثون الأصلي (نصي) إن طُلب ذلك
    """
    def __init__(self, رسالة, سطر=None, عمود=None, *, cause=None, include_traceback=False):
        self.رسالة = رسالة
        self.سطر = سطر
        self.عمود = عمود
        # snapshot of current context stack
        self._context = current_context()
        self._cause = cause
        self._py_tb = None
        if include_traceback and cause is not None:
            try:
                # format_exc will include the traceback of the latest exception
                self._py_tb = traceback.format_exc()
            except Exception:
                self._py_tb = None

        # construct base Exception message for Python ecosystems
        if سطر is not None and عمود is not None:
            super().__init__(f"{رسالة} (في السطر {سطر}، العمود {عمود})")
        elif سطر is not None:
            super().__init__(f"{رسالة} (في السطر {سطر})")
        else:
            super().__init__(رسالة)

    def with_source(self, مصدر: str):
        """
        إن وُفّر مصدر النص (string) والإحداثيات موجودة، ترجع رسالة موسعة
        تعرض السطر المعني وسهمًا يشير للعمود، مع سياق التنفيذ.
        مفيد لطباعة الأخطاء للمستخدم.
        """
        if self.سطر is None:
            return str(self)
        lines = مصدر.splitlines()
        if 1 <= self.سطر <= len(lines):
            line_text = lines[self.سطر - 1]
            col = self.عمود or 1
            if col < 1:
                col = 1
            if col > len(line_text) + 1:
                col = len(line_text) + 1
            pointer = ' ' * (col - 1) + '^'
            ctx = self.format_context()
            return f"{self.رسالة} (في السطر {self.سطر}، العمود {col})\n{line_text}\n{pointer}\n{ctx}"
        return str(self)

    def format_context(self):
        """
        إرجاع نص منسق يبيّن سياق التنفيذ (عناوين العقد/الناتجات) بترتيب
        يعرض الأحدث أولاً. إذا وُجد تتبُّع بايثون الأصلي يُدرج بعد السياق.
        """
        if not self._context:
            return ""
        lines = ["--- سياق تنفيذ (مقلوب، الأحدث أول):"]
        for i, c in enumerate(reversed(self._context), 1):
            lines.append(f"  {i}. {c}")
        if self._py_tb:
            lines.append("--- تتبُّع بايثون الأصلي ---")
            lines.append(self._py_tb)
        return "\n".join(lines)

    def __str__(self):
        base = self.رسالة
        if self.سطر is not None:
            base = f"{base} (سطر {self.سطر})"
        ctx = self.format_context()
        if ctx:
            return f"{base}\n{ctx}"
        return base
```

* أضفت تعليقات توضيحية (docstrings) لكل دالة لسهولة الصيانة.

# ملف: `iqra/interpreter.py`

هذا الملف هو قلب الـ **Alpha** — مفسّر (Interpreter) مُدقّق يدعم: زيارات لعقد AST، فتح/إغلاق بيئات (frames)، تعريف واستدعاء دوال المستخدم مع دعم **closures**، استدعاء غير مباشر (call indirect)، ورسائل خطأ محسّنة باستخدام `iqra.errors`.
الكود أدناه مُراجع ليكون واضحًا وقابلًا للتشغيل مع باقي ملفات الحزمة التي أرسلتها سابقًا.

```python
# -*- coding: utf-8 -*-
"""
مفسّر لغة "اقرأ" (Interpreter)
- يعتمد على عقد AST الموجودة في iqra.ast_nodes
- يستخدم iqra.errors لرسائل خطأ أفضل مع سياق التنفيذ
- يدعم builtins، تعريف الدوال واستدعائها، closures، واستدعاءات غير مباشرة.
"""

from typing import Any
from . import ast_nodes as ast
from .errors import خطأاقرأ, push_context, pop_context
from . import tokens
import random

class _ReturnSignal(Exception):
    """إشارة داخلية للتعامل مع عبارة 'ارجع' داخل الدوال."""
    def __init__(self, value: Any):
        self.value = value

class Interpreter:
    def __init__(self, builtins: dict = None):
        """
        frames: قائمة من القواميس تمثل الستاك البيئي (frames). frames[0] = global.
        functions: قاموس لتخزين دوال المستخدم بوصفها descriptors.
        builtins: دوال أصلية متاحة دائماً.
        """
        self.frames = [{}]  # stack of dicts; global is frames[0]
        self.functions = {}  # name -> descriptor {'type':'userfunc','ast':..., 'params':..., 'closure':env}
        # دوال أصلية مفيدة
        self.builtins = {
            'ادخل': lambda prompt='': input(prompt),
            'عشوائي': lambda a=0, b=1: random.randint(int(a), int(b)),
            'اطبع': lambda *args: print(*args),
        }
        if builtins:
            self.builtins.update(builtins)

    # ---- مساعدة على البيئة ----
    def current_env(self):
        return self.frames[-1]

    def push_env(self):
        self.frames.append({})

    def pop_env(self):
        if len(self.frames) == 1:
            # لا نزيل الإطار العالمي — نفضّه فقط
            self.frames[-1].clear()
        else:
            self.frames.pop()

    def resolve_var(self, name: str):
        """
        بحث عن اسم متغير/قيمة من أعلى الستاك إلى الأسفل.
        إن لم يُوجد، نتحقق من الدوال المسجَّلة ثم من builtins.
        """
        for env in reversed(self.frames):
            if name in env:
                return env[name]
        if name in self.functions:
            return self.functions[name]
        if name in self.builtins:
            return self.builtins[name]
        return None

    def store_var(self, name: str, val: Any):
        """
        يخزّن قيمة في أقرب إطار يحتوي هذا الاسم، وإلا يعينها في الإطار الحالي.
        هذا يسمح بتعديل متغيرات الإطار الخارجي في بعض السيناريوهات (سلوك مبسَّط).
        """
        for env in reversed(self.frames):
            if name in env:
                env[name] = val
                return
        self.current_env()[name] = val

    # ---- نقطة الدخول ----
    def interpret(self, node: ast.AST):
        """
        تدوير العقد: ندفع سياقًا قصيرًا ثم نستدعي معالج زيارة (visitor).
        جميع الأخطاء غير المتوقعة تُغلف بـ 'خطأاقرأ' مع تتبُّع بايثون الأصلي.
        """
        push_context(f"Node:{node.__class__.__name__}")
        try:
            method = getattr(self, f'visit_{node.__class__.__name__}', None)
            if method is None:
                raise خطأاقرأ(f'لا يوجد معالج للعقدة: {node.__class__.__name__}')
            return method(node)
        except خطأاقرأ:
            # أخطاء اللغة تخرج كما هي (مع السياق المرفوع سابقًا)
            raise
        except Exception as e:
            # نلف الاستثناء الأصلي داخل خطأاقرأ مع تتبُّع بايثون لحالات التصحيح
            raise خطأاقرأ(f"خطأ داخلي أثناء معالجة {node.__class__.__name__}: {e}", None, None, cause=e, include_traceback=True)
        finally:
            pop_context()

    # ---- زيارات لعقد قياسية ----
    def visit_Program(self, node: ast.Program):
        for stmt in node.statements:
            self.interpret(stmt)

    def visit_ExpressionStatement(self, node: ast.ExpressionStatement):
        return self.interpret(node.expression)

    def visit_PrintStatement(self, node: ast.PrintStatement):
        val = self.interpret(node.expression) if node.expression is not None else None
        # استدعاء builtin 'اطبع' إن وُجد
        printer = self.builtins.get('اطبع', print)
        return printer(val) if val is not None else printer()

    def visit_StringLiteral(self, node: ast.StringLiteral):
        return node.value

    def visit_NumberLiteral(self, node: ast.NumberLiteral):
        return node.value

    def visit_BooleanLiteral(self, node: ast.BooleanLiteral):
        return node.value

    def visit_ListLiteral(self, node: ast.ListLiteral):
        return [self.interpret(e) for e in node.elements]

    def visit_DictLiteral(self, node: ast.DictLiteral):
        out = {}
        for k_node, v_node in node.pairs:
            k = self.interpret(k_node)
            v = self.interpret(v_node)
            out[k] = v
        return out

    def _truthy(self, v: Any) -> bool:
        if isinstance(v, bool): return v
        if isinstance(v, (int, float)): return v != 0
        if isinstance(v, str): return v != ''
        if v is None: return False
        if isinstance(v, (list, dict)): return len(v) != 0
        return bool(v)

    # ---- عمليات ثنائية وأحادية ----
    def visit_BinaryOp(self, node: ast.BinaryOp):
        left = self.interpret(node.left)
        right = self.interpret(node.right)
        op = node.op

        if op == tokens.EQ: return left == right
        if op == tokens.NE: return left != right
        if op == tokens.LT: return left < right
        if op == tokens.GT: return left > right
        if op == tokens.LE: return left <= right
        if op == tokens.GE: return left >= right

        if op == tokens.AND: return self._truthy(left) and self._truthy(right)
        if op == tokens.OR:  return self._truthy(left) or self._truthy(right)

        if op == tokens.PLUS:
            # سلاسل أو قوائم
            if isinstance(left, list) and isinstance(right, list):
                return left + right
            if isinstance(left, str) or isinstance(right, str):
                return str(left) + str(right)
            return left + right
        if op == tokens.MINUS: return left - right
        if op == tokens.MULT: return left * right
        if op == tokens.DIV:
            if right == 0:
                raise خطأاقرأ('قسمة على صفر')
            return left / right

        raise خطأاقرأ(f'عامل ثنائي غير مدعوم: {op}')

    def visit_UnaryOp(self, node: ast.UnaryOp):
        val = self.interpret(node.operand)
        op = node.op
        if op == tokens.NOT:
            return not self._truthy(val)
        if op == tokens.MINUS:
            return -val
        raise خطأاقرأ(f'عامل أحادي غير مدعوم: {op}')

    # ---- متغيرات وفهارس ----
    def visit_VarAssign(self, node: ast.VarAssign):
        val = self.interpret(node.expression)
        self.store_var(node.name, val)
        return val

    def visit_VarReference(self, node: ast.VarReference):
        val = self.resolve_var(node.name)
        if val is None:
            raise خطأاقرأ(f'متغير أو دالة غير معرفة: {node.name}')
        return val

    def visit_Index(self, node: ast.Index):
        collection = self.interpret(node.collection)
        idx = self.interpret(node.index)
        if isinstance(collection, (list, tuple, str)):
            if not isinstance(idx, int):
                raise خطأاقرأ('الفهرس يجب أن يكون عدداً صحيحاً')
            n = len(collection)
            if idx < 0:
                idx = n + idx
            if idx < 0 or idx >= n:
                raise خطأاقرأ('فهرس خارج النطاق')
            return collection[idx]
        if isinstance(collection, dict):
            return collection.get(idx, None)
        raise خطأاقرأ('النوع غير قابل للفهرسة')

    def visit_SetItem(self, node: ast.SetItem):
        coll = self.interpret(node.collection)
        idx = self.interpret(node.index)
        val = self.interpret(node.expression)
        if isinstance(coll, list):
            if not isinstance(idx, int):
                raise خطأاقرأ('الفهرس يجب أن يكون عدداً صحيحاً')
            if idx < 0:
                idx = len(coll) + idx
            if idx < 0 or idx >= len(coll):
                raise خطأاقرأ('فهرس خارج النطاق')
            coll[idx] = val
            return val
        if isinstance(coll, dict):
            coll[idx] = val
            return val
        raise خطأاقرأ('النوع غير قابل للتعيين عبر فهرس')

    def visit_Slice(self, node: ast.Slice):
        collection = self.interpret(node.collection)
        start = self.interpret(node.start) if node.start is not None else None
        stop  = self.interpret(node.stop)  if node.stop is not None else None
        if isinstance(collection, (list, str)):
            return collection[slice(start, stop)]
        raise خطأاقرأ('القطع (slice) مدعوم فقط للقوائم والسلاسل')

    # ---- دوال ----
    def visit_FunctionDef(self, node: ast.FunctionDef):
        """
        تسجيل دالة المستخدم: نحفظ AST، المعاملات، ومرجع للبيئة الحالية كـ closure.
        ملاحظة: closure مخزّن كمرجع للإطار الحالي (ليس نسخة) — هذا مبسّط ويمكن تغييره لاحقًا.
        """
        func_obj = {
            'type': 'userfunc',
            'ast': node,
            'params': list(node.params),
            'closure': self.current_env(),
        }
        self.functions[node.name] = func_obj
        return None

    def visit_FunctionCall(self, node: ast.FunctionCall):
        """
        دعم الاستدعاء المباشر (اسم الدالة) أو الاستدعاء غير المباشر
        عبر تعبير (مثلاً: متغيرٍ يحمل دالة).
        """
        # تحديد القابل للاستدعاء
        callable_obj = None
        if isinstance(node.name, str):
            name = node.name
            # نبحث عن متغير بهذا الاسم (قد يخفي دالة مسجلة)
            val = None
            for env in reversed(self.frames):
                if name in env:
                    val = env[name]; break
            if val is not None:
                callable_obj = val
            elif name in self.builtins:
                callable_obj = self.builtins[name]
            elif name in self.functions:
                callable_obj = self.functions[name]
            else:
                raise خطأاقرأ(f'دالة أو متغير باسم {name} غير معرفة')
        else:
            # اسم ديناميكي (تعبير) => نقيّم التعبير
            callable_obj = self.interpret(node.name)

        # تقييم الوسائط
        args_vals = [self.interpret(a) for a in node.args]

        # إذا كان callable كائناً قابلاً للاستدعاء في بايثون (builtin أو قيمة بايثون)
        if callable(callable_obj) and not (isinstance(callable_obj, dict) and callable_obj.get('type') == 'userfunc'):
            try:
                return callable_obj(*args_vals)
            except TypeError as e:
                raise خطأاقرأ(f'خطأ في استدعاء الدالة الأصلية: {e}')

        # دالة مستخدم
        if isinstance(callable_obj, dict) and callable_obj.get('type') == 'userfunc':
            fn = callable_obj
            expected = len(fn.get('params', []))
            if len(args_vals) != expected:
                raise خطأاقرأ(f'عدد الوسائط في استدعاء {fn["ast"].name} غير صالح (المتوقع {expected}، المزوَّد {len(args_vals)})')
            # إذا وُجد closure مسجّل، نرفقه مؤقتًا إلى الستاك (يشير إلى إطار موجود)
            if fn.get('closure') is not None:
                self.frames.append(fn['closure'])
            # إطار محلي جديد للدالة
            self.push_env()
            for n, v in zip(fn.get('params', []), args_vals):
                self.current_env()[n] = v
            try:
                # تنفيذ جسم الدالة
                for s in fn['ast'].body:
                    self.interpret(s)
            except _ReturnSignal as r:
                # تنظيف الإطار المحلي وإرجاع القيمة
                self.pop_env()
                if fn.get('closure') is not None:
                    self.frames.pop()
                return r.value
            # لم يُرجع صراحة
            self.pop_env()
            if fn.get('closure') is not None:
                self.frames.pop()
            return None

        raise خطأاقرأ('الكائن المستدعى ليس دالة معروفة')

    def visit_ReturnStatement(self, node: ast.ReturnStatement):
        val = self.interpret(node.expression) if node.expression is not None else None
        raise _ReturnSignal(val)

    # ---- التحكم ----
    def visit_IfStatement(self, node: ast.IfStatement):
        cond = self.interpret(node.condition)
        if self._truthy(cond):
            for s in node.true_statements:
                self.interpret(s)
        else:
            for s in node.false_statements:
                self.interpret(s)

    def visit_WhileStatement(self, node: ast.WhileStatement):
        cond_val = self.interpret(node.condition)
        # تسهيلات: إن كان عددياً نفذ بعدد معين من المرات
        if isinstance(cond_val, (int, float)):
            for _ in range(int(cond_val)):
                for s in node.body:
                    self.interpret(s)
            return
        while self._truthy(self.interpret(node.condition)):
            for s in node.body:
                self.interpret(s)
```

## نقاط مهمة وتدقيقات تمّت في هذا الملف

* أضفت `اطبع` كـ builtin لتسهيل الاستخدام (يمكن تغييره أو حذفه إن أردت).
* إدارة الـ closures تتم ببساطة عن طريق إرفاق الـ closure environment (مرجع) مؤقتاً إلى `self.frames` عند استدعاء الدالة — هذا سلوك مبسَّط لكنه عملي. إن رغبت سأتبدّل إلى نسخ من البيئة (copy) إذا أردت سلوكًا أقوى لعزل المتغيرات.
* كل استثناء غير متوقع يُحاط بـ `خطأاقرأ` مع `include_traceback=True` لمساعدة المطور عند التصحيح.
* الشيفرة تفترض وجود تعريفات العقد (AST classes) في `iqra.ast_nodes` مع أسماء مثل `Program`, `FunctionDef`, `FunctionCall`, `VarAssign`, إلخ. تأكد أن `ast_nodes` مُطابقة لهذا التوقيع (سأعرض الملف التالي إذا ترغب).







نبدأ بـ **`iqra/ast_nodes.py`** — وهو ملف مهم لأنه يعرّف كل عقد الـAST التي يستخدمها الـInterpreter و الـCompiler و الـVM.

```python
# -*- coding: utf-8 -*-
"""
Lexer للغة "اقرأ": يحول النص المصدري إلى قائمة Tokens.
"""

import re
from . import tokens
from .errors import خطأاقرأ

class Token:
    def __init__(self, type_, value, line, column):
        self.type = type_
        self.value = value
        self.line = line
        self.column = column
    def __repr__(self):
        return f"Token({self.type}, {self.value}, {self.line}:{self.column})"

class Lexer:
    def __init__(self, source):
        self.source = source
        self.position = 0
        self.line = 1
        self.column = 1

    def tokenize(self):
        tokens_list = []
        while not self._is_at_end():
            char = self._peek()
            if char.isspace():
                self._advance_whitespace()
                continue
            if char.isdigit():
                tokens_list.append(self._number())
                continue
            if char.isalpha() or char == '_':
                tokens_list.append(self._identifier_or_keyword())
                continue
            if char in ('"', "'"):
                tokens_list.append(self._string())
                continue
            # الرموز المفردة والمزدوجة
            two_char = self._peek(2)
            if two_char in (tokens.EQ, tokens.NE, tokens.LE, tokens.GE):
                tokens_list.append(Token(two_char, two_char, self.line, self.column))
                self._advance(2)
                continue
            if char in (tokens.PLUS, tokens.MINUS, tokens.MULT, tokens.DIV,
                        tokens.LT, tokens.GT, '=', '(', ')', '{', '}', ',', ':', '[', ']'):
                tokens_list.append(Token(char, char, self.line, self.column))
                self._advance()
                continue
            raise خطأاقرأ(f"رمز غير معروف: {char}", self.line, self.column)
        return tokens_list

    def _is_at_end(self):
        return self.position >= len(self.source)

    def _peek(self, length=1):
        if self.position + length > len(self.source):
            return self.source[self.position:]
        return self.source[self.position:self.position+length]

    def _advance(self, n=1):
        for _ in range(n):
            if self._is_at_end():
                return
            if self.source[self.position] == '\n':
                self.line += 1
                self.column = 1
            else:
                self.column += 1
            self.position += 1

    def _advance_whitespace(self):
        while not self._is_at_end() and self._peek().isspace():
            self._advance()

    def _number(self):
        start_line, start_col = self.line, self.column
        num_str = ''
        while not self._is_at_end() and (self._peek().isdigit() or self._peek() == '.'):
            num_str += self._peek()
            self._advance()
        return Token('NUMBER', float(num_str), start_line, start_col)

    def _identifier_or_keyword(self):
        start_line, start_col = self.line, self.column
        ident = ''
        while not self._is_at_end() and (self._peek().isalnum() or self._peek() == '_'):
            ident += self._peek()
            self._advance()
        if ident in tokens.KEYWORDS:
            return Token(tokens.KEYWORDS[ident], ident, start_line, start_col)
        return Token('IDENT', ident, start_line, start_col)

    def _string(self):
        quote = self._peek()
        start_line, start_col = self.line, self.column
        self._advance()  # skip quote
        val = ''
        while not self._is_at_end() and self._peek() != quote:
            val += self._peek()
            self._advance()
        if self._is_at_end():
            raise خطأاقرأ("سلسلة نصية غير منتهية", start_line, start_col)
        self._advance()  # skip closing quote
        return Token('STRING', val, start_line, start_col)
```

```python
# -*- coding: utf-8 -*-
"""
Parser للغة "اقرأ": يحوّل قائمة Tokens إلى شجرة AST.
"""
from . import tokens
from .errors import خطأاقرأ
from . import ast_nodes as ast

class Parser:
    def __init__(self, tokens_list):
        self.tokens = tokens_list
        self.position = 0

    def parse(self):
        statements = []
        while not self._is_at_end():
            statements.append(self._statement())
        return ast.Program(statements)

    def _is_at_end(self):
        return self.position >= len(self.tokens)

    def _peek(self):
        return self.tokens[self.position]

    def _advance(self):
        if not self._is_at_end():
            self.position += 1

    def _match(self, *types):
        if self._is_at_end():
            return False
        if self._peek().type in types:
            self._advance()
            return True
        return False

    def _consume(self, type_, msg):
        if self._match(type_):
            return self.tokens[self.position-1]
        raise خطأاقرأ(msg)

    def _statement(self):
        tok = self._peek()
        if tok.type == 'PRINT':
            self._advance()
            expr = self._expression()
            return ast.PrintStatement(expr)
        # TODO: Add parsing for other statements (if, while, func, etc.)
        expr = self._expression()
        return ast.ExpressionStatement(expr)

    def _expression(self):
        return self._primary()

    def _primary(self):
        tok = self._peek()
        if tok.type == 'NUMBER':
            self._advance()
            return ast.NumberLiteral(tok.value)
        if tok.type == 'STRING':
            self._advance()
            return ast.StringLiteral(tok.value)
        if tok.type == 'IDENT':
            self._advance()
            return ast.VarReference(tok.value)
        raise خطأاقرأ(f"تعبير غير متوقع: {tok.type}")
```

```python
# -*- coding: utf-8 -*-
"""
Compiler للغة "اقرأ": يحوّل شجرة AST إلى بايتكود بسيط قابل للتنفيذ في VM.
"""

from . import ast_nodes as ast
from .errors import خطأاقرأ

class Compiler:
    def __init__(self):
        self.code = []     # قائمة التعليمات (op, args...)
        self.consts = []   # القيم الثابتة
        self.varnames = [] # أسماء المتغيرات المستخدمة

    def compile(self, node):
        method = getattr(self, f'compile_{node.__class__.__name__}', None)
        if not method:
            raise خطأاقرأ(f"لا يوجد مترجم لهذه العقدة: {node.__class__.__name__}")
        method(node)

    def add_const(self, value):
        if value in self.consts:
            return self.consts.index(value)
        self.consts.append(value)
        return len(self.consts) - 1

    # عقد أساسية
    def compile_Program(self, node: ast.Program):
        for stmt in node.statements:
            self.compile(stmt)

    def compile_PrintStatement(self, node: ast.PrintStatement):
        self.compile(node.expression)
        self.code.append(('PRINT',))

    def compile_ExpressionStatement(self, node: ast.ExpressionStatement):
        self.compile(node.expression)
        self.code.append(('POP_TOP',))

    def compile_NumberLiteral(self, node: ast.NumberLiteral):
        idx = self.add_const(node.value)
        self.code.append(('PUSH_CONST', idx))

    def compile_StringLiteral(self, node: ast.StringLiteral):
        idx = self.add_const(node.value)
        self.code.append(('PUSH_CONST', idx))

    def compile_VarReference(self, node: ast.VarReference):
        self.code.append(('LOAD_VAR', node.name))

    def compile_VarAssign(self, node: ast.VarAssign):
        self.compile(node.expression)
        self.code.append(('STORE_VAR', node.name))
```

```python
# -*- coding: utf-8 -*-
"""
VM (آلة افتراضية) للغة "اقرأ": تنفّذ البايتكود الناتج عن الـCompiler.
"""

from .errors import خطأاقرأ

class VM:
    def __init__(self, code, consts, globals_vars=None, builtins=None, enable_jit=False):
        self.code = code
        self.consts = consts
        self.stack = []
        self.vars = globals_vars if globals_vars is not None else {}
        self.builtins = builtins if builtins is not None else {}
        self.enable_jit = enable_jit
        self.ip = 0  # instruction pointer

    def push(self, val):
        self.stack.append(val)

    def pop(self):
        return self.stack.pop()

    def run(self):
        while self.ip < len(self.code):
            instr = self.code[self.ip]
            self.ip += 1
            op = instr[0]

            if op == 'PUSH_CONST':
                idx = instr[1]
                self.push(self.consts[idx])
            elif op == 'LOAD_VAR':
                name = instr[1]
                if name in self.vars:
                    self.push(self.vars[name])
                elif name in self.builtins:
                    self.push(self.builtins[name])
                else:
                    raise خطأاقرأ(f'متغير غير معروف: {name}')
            elif op == 'STORE_VAR':
                name = instr[1]
                val = self.pop()
                self.vars[name] = val
            elif op == 'PRINT':
                val = self.pop()
                print(val)
            elif op == 'POP_TOP':
                self.pop()
            else:
                raise خطأاقرأ(f"تعليمة غير معروفة: {op}")
```

```python
"""
REPL تفاعلي للغة "اقرأ".
"""

from .lexer import Lexer
from .parser import Parser
from .interpreter import Interpreter
from .compiler import Compiler
from .vm import VM
from .errors import خطأاقرأ

def repl_loop():
    print("مرحبًا بك في REPL للغة اقرأ. اكتب ':quit' للخروج.")
    interp = Interpreter()
    buffer = []
    while True:
        try:
            line = input(">>> " if not buffer else "... ")
        except EOFError:
            break
        if line.strip() == ':quit':
            break
        if not line.strip():
            if buffer:
                source = '\n'.join(buffer)
                try:
                    lexer = Lexer(source)
                    toks = lexer.tokenize()
                    parser = Parser(toks)
                    program = parser.parse()
                    interp.interpret(program)
                except خطأاقرأ as e:
                    print("خطأ:", e)
                buffer = []
            continue
        buffer.append(line)
```
